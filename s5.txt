#include<bits/stdc++.h>
#define double long double
using namespace std;
#define PI acos(-1)
int main(){
#define int long long int 

	int t;
	cin>>t;
	while(t--)
	{
		double R,r,p,x,t;
		cin>>R>>r>>p>>x>>t;          //p=v/PI

		double t1=(2*(R-r))/p;
		double t2=(2*(R+r))/p;
		
//		cout<<fixed<<setprecision(6)<<t1<<' '<<fixed<<setprecision(6)<<t2<<endl;
	

		double T=t1+t2;
	

		int full_revolution=(t/T);
		int xx=full_revolution;
		full_revolution*=2;
		
//		cout<<full_revolution<<endl;
		

		double left=t-xx*T;
		double angle=0.0000000000;
		
	//	cout<<fixed<<setprecision(6)<<left<<endl;
		
		
		

		if(left>0)
		{
			if(left==t1)
			{
				full_revolution++;
				left=0;
			}

			else if(left<t1)
			{
				angle=(360/t1)*left;

			}
			else
			{
				full_revolution++;
				left-=t1;
				angle=360-(360/t2)*left;
			}

		}

		double full=full_revolution*x;

	



		double ans_angle=(angle+full);

		int py=(ans_angle/360);

		ans_angle=ans_angle-py*360;
			
			



		int flag;

		if(full_revolution & 1)
			flag=1;
		else
			flag=0;

		cout<<flag<<' '<<fixed<<setprecision(9)<<ans_angle<<endl;


	}
return 0;
}
