#include<bits/stdc++.h>
using namespace std;
int main(){
#define int long long int 
 
int t;
cin>>t;
while(t--)
{
	int a,b;
	cin>>a>>b;
 
	double x=((2*a+2*b)+sqrt(4*(a+b)*(a+b)-12*a*b))/6;
	double y=((2*a+2*b)-sqrt(4*(a+b)*(a+b)-12*a*b))/6;
 
	x+=.5;
	y+=.5;
 
	int xx=x;
	int yy=y;
 
	int vx=xx*(a-xx)*(b-xx);
	int vy=yy*(a-yy)*(b-yy);
 
	if(vx>vy)
	{
		cout<<xx<<' '<<vx<<endl;
 
	}
	else
		cout<<yy<<' '<<vy<<endl;
}
}
