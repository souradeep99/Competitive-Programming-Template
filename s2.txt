#include<bits/stdc++.h>
using namespace std;
int main(){
#define int long long int 

int n;
cin>>n;

int a[n];

for (int i = 0; i < n; ++i)
{
	cin>>a[i];
}

int lo=-1,hi=-1,maxi=0,p=0,loo=-1,hoo=-1;

for(int i=0;i<n;i++)
{

	if(a[i]==0)
	{
		maxi=max(maxi,p);

		if(maxi>p)
		{
			loo=lo;
			hoo=hi;
		}

		lo=-1;
		hi=-1;
		p=0;
	}

	else if(a[i]!=0 && lo==-1)
	{
		lo=i;
	}
	else if(a[i]!=0 && lo!=-1)
	{
		p++;
		hi=i;
	}

}

if(maxi==0)
{
maxi=p;
loo=lo;
hoo=hi;
}

//cout<<hoo<<' '<<loo<<endl;



int seg[hoo-loo+1],j=0;
int size=hoo-loo+1;
for (int i = loo; i <= hoo; ++i)
{
	seg[j]=a[i];
	j++;
}

sort(seg,seg+size);
int ans=1;

for (int i = 1; i < size; ++i)
{
	if(seg[i]!=seg[i-1])
		ans++;
}

cout<<ans<<endl;
}
